pipeline {
    agent any

    environment {
        IMAGE_NAME = "apache-web"
        IMAGE_TAG  = "v1"
    }

    stages {
        stage('Checkout Code') {
            steps {
                git branch: 'main',
                    url: 'https://github.com/your-repo/apache-docker-project.git'
            }
        }

        stage('Build Docker Image') {
            steps {
                sh '''
                docker build -t $IMAGE_NAME:$IMAGE_TAG .
                '''
            }
        }

        stage('Verify Image') {
            steps {
                sh 'docker images | grep apache-web'
            }
        }
    }
}
